（二）基本思想

#定义微分方程 我这里用的是美赛的一道题的微分方程 注意微分方程的规范性
def equal(x,y):
    l = ((1 / u) * np.exp(-1 * u * x) * ((np.exp(u * x)) - 1) * (vf + sigama * b * np.sin(angle) * vs * np.sin(angle)))
    return a * b * vs * l * np.cos(angle) - 0.0016 * y
#定义一个函数
def fun(x):
    return np.pi*x*x

2.人脑分析问题

对于数据分析的问题。要明白自己要研究什么：

比如你研究利润和营业时间的关系那就关注营业时间而不是关注你的商品能不能一半放生一半拿来喂鱼。

找到研究的对象后，可能要对研究的对象进行数据处理，减少程序处理数据的时间。

对于“非黑即白之类”的分类问题可以把“是否”用0和1来表示；

比如营业时间这种离散程度很大的指标，可以划分区间进行研究，比如以几分钟作为一个观察区间。

import numpy as np

sport=pd.read_csv(r'G:\Python-Data-Aanalysis-and-Miner\data\Run or Walk.csv')

#多元回归（岭回归）
import numpy as np
import  matplotlib.pyplot as plt
from scipy import integrate
import statsmodels.api as sm
import pandas as pd
import seaborn as sns
from scipy.stats import f
from sklearn import model_selection
from sklearn.linear_model import Ridge,RidgeCV
plt.rcParams['font.sans-serif']=['Microsoft YaHei']
plt.rcParams['axes.unicode_minus']=False
plt.style.use('ggplot')
data=pd.read_excel(r'G:\Python-Data-Aanalysis-and-Miner\data\diabetes.xlsx')
'''
可视化方法确定lambda值得范围
'''
predictors=data.columns[2:-1]
x_train,x_test,y_train,y_test=model_selection.train_test_split(data[predictors],data['Y'],test_size=0.2,random_state=1234)
lambdas=np.logspace(-5,2,200)
'''logspace是用来构建等比数列 以10^-5开始直到10^5结束，分割成200个'''
ridge_cofficients=[]
for i in lambdas:
    ridge=Ridge(alpha=i,normalize=True)
    ridge.fit(x_train,y_train)
    ridge_cofficients.append(ridge.coef_)
plt.plot(lambdas,ridge_cofficients)
plt.xscale('log')

''''’‘’‘’‘’‘’‘’‘’‘’''''''''''''''''''''''''''''''''
''''''''''''''''''交叉验证法确定lambda值'''''''''''''
'''''''''''''''''''''''''''''''''''''''''''''''''''

ridge_cv=RidgeCV(alphas=lambdas,normalize=True,scoring='neg_mean_squared_error',cv=10)
ridge_cv.fit(x_train,y_train)
ridge_best_lambda=ridge_cv.alpha_
print(ridge_best_lambda)


ridge=Ridge(alpha=ridge_best_lambda,normalize=True)
ridge.fit(x_train,y_train)
print(pd.Series(index=['Intercept']+x_train.columns.tolist(),data=[ridge.intercept_]+ridge.coef_.tolist()))
